// --- კონფიგურაცია --- const myId = localStorage.getItem('user_id'); const myNick = localStorage.getItem('nickname') || "სტუმარი"; const roomId = "main_table_001"; // მაგიდის ID // --- 1. თამაშის დაწყება და პროფილის სინქრონიზაცია --- async function initGame() { const { data: profile } = await _s.from('profiles').select('*').eq('id', myId).single(); // ეკრანზე შენი მონაცემების დასმა (თხის ნაცვლად) document.getElementById('my-nick-display').innerText = profile.username; document.getElementById('my-avatar').src = profile.avatar_url || 'default.png'; // მაგიდაზე დაჯდომა const { data: game } = await _s.from('joker_games').select('*').eq('room_id', roomId).single(); let players = game.players || []; if (!players.find(p => p.id === myId) && players.length < 4) { players.push({ id: myId, name: profile.username, avatar: profile.avatar_url, hand: [], // ხელში არსებული კარტები bid: 0, // რამდენი თქვა taken: 0, // რამდენი წაიღო points: 0 // ჯამური ქულა }); await _s.from('joker_games').update({ players }).eq('room_id', roomId); } renderTable(players, game.table_cards, game.turn_index); } // --- 2. კარტების აჩეხვა და დარიგება (მხოლოდ "Abu007"-ს შეუძლია) --- async function dealCards() { if (myNick !== "Abu007") return alert("მხოლოდ ადმინს შეუძლია დარიგება!"); const deck = createDeck(); // ქმნის 52 კარტს + 2 ჯოკერი const { data: game } = await _s.from('joker_games').select('players').eq('room_id', roomId).single(); let players = game.players; // თითოს 9 კარტი (მაგალითად პირველი ხელისთვის) players.forEach(p => { p.hand = deck.splice(0, 9); p.bid = 0; p.taken = 0; }); await _s.from('joker_games').update({ players, deck, table_cards: [], turn_index: 0, status: 'bidding' }).eq('room_id', roomId); } function createDeck() { const suits = ['❤️', '♦️', '♣️', '♠️']; const values = ['6','7','8','9','10','J','Q','K','A']; let deck = []; suits.forEach(s => values.forEach(v => deck.push({ suit: s, val: v }))); deck.push({ suit: 'JOKER', val: '🃏' }, { suit: 'JOKER', val: '🃏' }); return deck.sort(() => Math.random() - 0.5); // არევა } // --- 3. ვიზუალური მხარე (მაგიდის დახატვა) --- function renderTable(players, tableCards, turnIndex) { // მოთამაშეების დასმა for (let i = 1; i <= 4; i++) { const p = players[i-1]; const slotName = document.getElementById(`p${i}_name`); const slotImg = document.getElementById(`p${i}_img`); if (p) { slotName.innerText = p.name + (i-1 === turnIndex ? " 🟢" : ""); slotImg.src = p.avatar || 'default.png'; // თუ მე ვარ, ჩემი კარტები გამომიჩინოს if (p.id === myId) renderMyHand(p.hand); } else { slotName.innerText = "ელოდება..."; slotImg.src = "placeholder.png"; } } // მაგიდაზე დაგდებული კარტების ჩვენება const tableDiv = document.getElementById('table-center'); tableDiv.innerHTML = ""; (tableCards || []).forEach(c => { tableDiv.innerHTML += `
${c.val}${c.suit}
`; }); } function renderMyHand(hand) { const handDiv = document.getElementById('my-cards'); handDiv.innerHTML = ""; hand.forEach((card, index) => { const c = document.createElement('div'); c.className = 'card clickable'; c.innerHTML = `${card.val}${card.suit}`; c.onclick = () => playCard(index); handDiv.appendChild(c); }); } // --- 4. კარტის დაგდება --- async function playCard(cardIndex) { const { data: game } = await _s.from('joker_games').select('*').eq('room_id', roomId).single(); // შემოწმება: ჩემი ჯერია? const myIndex = game.players.findIndex(p => p.id === myId); if (game.turn_index !== myIndex) return alert("შენი ჯერი არაა!"); let players = game.players; let table = game.table_cards || []; const card = players[myIndex].hand.splice(cardIndex, 1)[0]; table.push({ ...card, owner: myNick }); let nextTurn = (game.turn_index + 1) % players.length; await _s.from('joker_games').update({ players, table_cards: table, turn_index: nextTurn }).eq('room_id', roomId); } // --- 5. ჩატი და წაშლა (Admin Abu007) --- async function handleChat() { const { data: msgs } = await _s.from('messages').select('*').eq('room_name', 'joker_chat').limit(20).order('created_at', {ascending: false}); const chatBox = document.getElementById('joker-chat-box'); chatBox.innerHTML = msgs.reverse().map(m => `
${m.username}: ${m.text} ${(myNick === 'Abu007' || m.user_id === myId) ? `` : ''}
`).join(''); chatBox.scrollTop = chatBox.scrollHeight; } async function deleteMsg(id) { await _s.from('messages').delete().eq('id', id); } // --- 6. Realtime (ავტომატური განახლება ყველასთვის) --- _s.channel('game_channel') .on('postgres_changes', { event: '*', schema: 'public', table: 'joker_games' }, payload => { renderTable(payload.new.players, payload.new.table_cards, payload.new.turn_index); }) .on('postgres_changes', { event: 'INSERT', schema: 'public', table: 'messages' }, () => handleChat()) .subscribe(); // გაშვება window.onload = () => { initGame(); handleChat(); };
